/**
 * Fix Custom Setting Page Layout Assignment
 * 
 * This script checks the current state of page layout assignments for the
 * NZC_EmissionsMatchingConfig__c Custom Setting and provides guidance.
 * 
 * INSTRUCTIONS:
 * 1. Deploy the page layout file first:
 *    - File: force-app/main/default/layouts/NZC_EmissionsMatchingConfig__c-NZC Emissions Matching Configuration Layout.layout
 *    - Run: sf project deploy start --source-dir force-app/main/default/layouts
 * 
 * 2. Then run this script to check the current state
 * 
 * 3. Follow the manual steps below to assign the layout to all profiles
 */

System.debug('=== Custom Setting Page Layout Check ===\n');

String objectApiName = 'NZC_EmissionsMatchingConfig__c';
String layoutName = 'NZC Emissions Matching Configuration Layout';

try {
    // Check if Custom Setting exists
    Schema.SObjectType objType = Schema.getGlobalDescribe().get(objectApiName);
    if (objType == null) {
        System.debug('✗ Custom Setting "' + objectApiName + '" not found.');
        System.debug('  Please deploy the main codebase first.');
        return;
    }
    System.debug('✓ Custom Setting "' + objectApiName + '" exists.\n');

    // Check profiles (layout assignment must be done manually)
    List<Profile> profiles = [SELECT Id, Name FROM Profile WHERE UserType = 'Standard' LIMIT 10];
    System.debug('Found ' + profiles.size() + ' standard profiles (showing first 10).\n');
    System.debug('Note: Layout metadata cannot be queried via SOQL. Please verify layout deployment in Setup UI.\n');

    System.debug('\n=== MANUAL STEPS REQUIRED ===');
    System.debug('Page layout assignment must be done manually via Setup UI:\n');
    System.debug('OPTION 1: Assign via Object Manager (Recommended)');
    System.debug('1. Go to Setup > Object Manager > NZC Emissions Matching Configuration');
    System.debug('2. Click "Page Layouts" in the left sidebar');
    System.debug('3. Click "Page Layout Assignment" button');
    System.debug('4. For each profile, select the layout: "' + layoutName + '"');
    System.debug('5. Click "Save"\n');
    
    System.debug('OPTION 2: Assign via Profile');
    System.debug('1. Go to Setup > Users > Profiles');
    System.debug('2. Click on a profile name');
    System.debug('3. Scroll down to "Page Layout Assignment" section');
    System.debug('4. Find "NZC Emissions Matching Configuration"');
    System.debug('5. Select the layout: "' + layoutName + '"');
    System.debug('6. Click "Save"');
    System.debug('7. Repeat for all profiles\n');
    
    System.debug('OPTION 3: Use Metadata API (Advanced)');
    System.debug('1. Retrieve profile metadata:');
    System.debug('   sf project retrieve start --metadata Profile:Admin');
    System.debug('2. Edit the profile XML to add layout assignment');
    System.debug('3. Deploy back: sf project deploy start\n');
    
    System.debug('=== QUICK FIX FOR TESTING ===');
    System.debug('If you just need to view the Custom Setting record for testing:');
    System.debug('1. Go to Setup > Custom Settings');
    System.debug('2. Click "NZC Emissions Matching Configuration"');
    System.debug('3. Click "Manage" (this will work even without layout assignment)');
    System.debug('4. The error occurs when trying to view individual records\n');
    
    System.debug('=== VERIFICATION ===');
    System.debug('After assigning layouts, verify by:');
    System.debug('1. Going to Setup > Custom Settings > NZC Emissions Matching Configuration');
    System.debug('2. Clicking "Manage"');
    System.debug('3. Clicking on a record (should not show error)\n');

} catch (Exception e) {
    System.debug('✗ Error: ' + e.getMessage());
    System.debug('Stack Trace: ' + e.getStackTraceString());
}
